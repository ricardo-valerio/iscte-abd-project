    For each country:

    1. Number of Players
    2. Number of Tournaments
    3. Number of Games

    
                      SELECT
                          country.country AS country_name,
                          (SELECT COUNT(*)
                           FROM player
                           WHERE player.born_country_code = country.country_code) AS TotalPlayers,
                          (SELECT COUNT(*)
                           FROM tournament
                           WHERE tournament.location_country_code = country.country_code) AS TotalTournaments,
                          (SELECT COUNT(*)
                           FROM match
                           JOIN tournament ON match.tournament_id = tournament.id
                           WHERE tournament.location_country_code = country.country_code) AS TotalGames
                      FROM
                          country;
