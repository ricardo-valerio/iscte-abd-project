    List of Top 5 players name with their winning games in Hard Ground.

    
                SELECT
                    player.name AS player_name,
                    COUNT(player.id) AS winning_games
                FROM
                    player
                LEFT JOIN
                    match ON player.id = match.winner_id
                WHERE
                    match.ground = 'Hard'
                GROUP BY
                    player.id
                ORDER BY
                    winning_games DESC
                LIMIT 5;
