    List of Top 10 left-handed players name with their % of winning games in Grand Slam games.

    
                SELECT A.Name,
                       Wins,
                       Losses,
                       (Wins + Losses) AS TotalGames,
                       ROUND((CAST(Wins AS DECIMAL) / (Wins + Losses) * 100), 2) AS WinRate
                FROM
                    (SELECT player.name AS Name, COUNT(*) AS Wins
                     FROM player
                     JOIN match ON player.id = match.winner_id
                     JOIN tournament ON match.tournament_id = tournament.id
                     WHERE player.forehand = 'Left-Handed'
                     AND tournament.name IN ('Australian Open', 'Roland Garros', 'Wimbledon', 'US Open')
                     GROUP BY player.name) AS A
                JOIN
                    (SELECT player.name AS Name, COUNT(*) AS Losses
                     FROM player
                     JOIN match ON player.id = match.loser_id
                     JOIN tournament ON match.tournament_id = tournament.id
                     WHERE player.forehand = 'Left-Handed'
                     AND tournament.name IN ('Australian Open', 'Roland Garros', 'Wimbledon', 'US Open')
                     GROUP BY player.name) AS B
                ON A.Name = B.Name
                HAVING TotalGames >= 100
                ORDER BY WinRate DESC
                LIMIT 10;
